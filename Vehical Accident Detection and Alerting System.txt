#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <TinyGPS++.h> // Library for parsing GPS data
#include <SoftwareSerial.h> // For GPS communication

// Motor driver pins
const int ENA = 5;
const int IN1 = 6;
const int IN2 = 7;
const int ENB = 10;
const int IN3 = 9;
const int IN4 = 8;

// Relay and switch pins
const int relayPin = 4;
const int switchPin = 3;
const int startButtonPin = 11; // Start button pin
const int buzzerPin = 2;       // <<< ADDED: Define buzzer pin

// ADXL335 pins
const int xPin = A0;
const int yPin = A1;
const int zPin = A2;

// Variables for ADXL335 tilt detection
const int maxTiltThreshold = 55;
int normalX = 0, normalY = 0, normalZ = 0;

// GSM SIM800L settings
const String phoneNumber = "+918767793279";
const String alertMessage = "Accident detected! Please check immediately. Sending Location";
bool systemStopped = false;
bool systemReadyToStart = false; // Variable to track if the system is ready to start

// Variable to store speed (0-255)
int speed = 60;

// LCD setup
LiquidCrystal_I2C lcd(0x27, 16, 2);

// GPS setup
TinyGPSPlus gps;
SoftwareSerial gpsSerial(12, 13); // GPS TX pin to D8, RX pin to D9

void setup() {
  Serial.begin(9600); // Hardware serial for GSM
  gpsSerial.begin(9600); // Serial communication with GPS

  lcd.init();
  lcd.backlight();
  lcd.setCursor(0, 0);
  lcd.print("Initializing...");
  delay(2000);

  // Initialize motor pins as outputs
  pinMode(ENA, OUTPUT);
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(ENB, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);

  // Initialize relay, switch, and start button
  pinMode(relayPin, OUTPUT);
  pinMode(switchPin, INPUT_PULLUP);
  pinMode(startButtonPin, INPUT_PULLUP);
  pinMode(buzzerPin, OUTPUT);      // <<< ADDED
  digitalWrite(buzzerPin, LOW);  // <<< ADDED: Ensure buzzer is off at start

  // Calibrate ADXL335
  lcd.clear();
  lcd.print("Calibrating...");
  delay(5000); // Allow time to stabilize
  normalX = getAverageReading(xPin);
  normalY = getAverageReading(yPin);
  normalZ = getAverageReading(zPin);

  lcd.clear();
  lcd.print("System Ready");
  Serial.println("System Ready. Press Start Button to begin.");
}

void loop() {
  if (!systemReadyToStart) {
    // Wait for start button press
    if (digitalRead(startButtonPin) == LOW) {
      lcd.clear();
      lcd.print("Starting...");
      delay(2000);
      systemReadyToStart = true; // Allow the system to start
      lcd.clear();
      lcd.print("System Active");
    }
    return; // Do not proceed until start button is pressed
  }

  if (systemStopped) {
    sendLiveLocation(); // Periodically send live location updates
    return;
  }

  // Check the state of the switch
  bool isSwitchPressed = digitalRead(switchPin) == LOW;

  // Read current accelerometer values
  int xVal = analogRead(xPin);
  int yVal = analogRead(yPin);
  int zVal = analogRead(zPin);

  // Calculate deviations
  int xDeviation = abs(xVal - normalX);
  int yDeviation = abs(yVal - normalY);
  int zDeviation = abs(zVal - normalZ);

  // Detect maximum tilt
  bool isMaxTilt = (xDeviation > maxTiltThreshold || yDeviation > maxTiltThreshold || zDeviation > maxTiltThreshold);

  if (isSwitchPressed || isMaxTilt) {
    handleAccident();
    systemStopped = true;
  } else {
    // Normal operation
    digitalWrite(IN1, HIGH);
    digitalWrite(IN2, LOW);
    digitalWrite(IN3, HIGH);
    digitalWrite(IN4, LOW);

    // Set the speed
    analogWrite(ENA, speed);
    analogWrite(ENB, speed);

    // Turn off the relay
    digitalWrite(relayPin, LOW);
    digitalWrite(buzzerPin, LOW); // <<< ADDED: Keep buzzer off during normal operation
    lcd.clear();
    lcd.print("Motors Running");
  }

  delay(200); // Small delay for stable readings
}

void handleAccident() {
  // Stop motors
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, LOW);
  analogWrite(ENA, 0);
  analogWrite(ENB, 0);

  // Activate relay
  digitalWrite(relayPin, HIGH);
  digitalWrite(buzzerPin, HIGH); // <<< ADDED: Turn on the buzzer

  // Send accident alert SMS
  sendAccidentAlert();
}

void sendAccidentAlert() {
  String location = getGPSLocation();
  String message = alertMessage;
  if (location.length() > 0) {
    message += ". Location: " + location;
  }

  Serial.println("[INFO] Sending accident alert SMS...");
  lcd.clear();
  lcd.print("Sending SMS...");
  sendSMS(message);

  Serial.println("[INFO] Accident alert SMS sent.");
  lcd.clear();
  lcd.print("SMS Sent");
}

String getGPSLocation() {
  while (gpsSerial.available() > 0) {
    if (gps.encode(gpsSerial.read())) {
      if (gps.location.isValid()) {
        String latitude = String(gps.location.lat(), 6);
        String longitude = String(gps.location.lng(), 6);
        return "https://maps.google.com/?q=" + latitude + "," + longitude;
      }
    }
  }
  return ""; // Return empty if location is not available
}

void sendLiveLocation() {
  String location = getGPSLocation();
  if (location.length() > 0) {
    String liveLocationMessage = "Live Location: " + location;
    Serial.println("[INFO] Sending live location SMS...");
    lcd.clear();
    lcd.print("Live Loc SMS...");
    sendSMS(liveLocationMessage);
    lcd.clear();
    lcd.print("Loc Sent");
    delay(10000); // Send updates every 10 seconds
  }
}

void sendSMS(String message) {
  Serial.println("AT");
  delay(1000);
  Serial.println("AT+CMGF=1"); // Set SMS to text mode
  delay(1000);
  Serial.print("AT+CMGS=\"");
  Serial.print(phoneNumber);
  Serial.println("\"");
  delay(1000);
  Serial.print(message);
  Serial.write(26); // Ctrl+Z to send SMS
  delay(5000);
}

int getAverageReading(int pin) {
  long total = 0;
  const int samples = 50;
  for (int i = 0; i < samples; i++) {
    total += analogRead(pin);
    delay(10);
  }
  return total / samples;
}
